Database Schema

-- Core Tables CREATE TABLE users ( id SERIAL PRIMARY KEY, email VARCHAR(255) UNIQUE, password_hash VARCHAR(255), role ENUM('CUSTOMER', 'MANAGER', 'FLORIST', 'DRIVER'), loyalty_points INTEGER DEFAULT 0 ); CREATE TABLE raw_materials ( id SERIAL PRIMARY KEY, name VARCHAR(100), stock_quantity INTEGER, unit VARCHAR(50), low_stock_threshold INTEGER ); CREATE TABLE products ( id SERIAL PRIMARY KEY, name VARCHAR(100), price DECIMAL(10,2), is_addon BOOLEAN DEFAULT FALSE, category VARCHAR(50) ); CREATE TABLE product_recipes ( id SERIAL PRIMARY KEY, product_id INTEGER REFERENCES products(id), raw_material_id INTEGER REFERENCES raw_materials(id), quantity_required INTEGER );
